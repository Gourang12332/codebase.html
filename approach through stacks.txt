import java.util.* ;



import java.io.*; 
public class Solution {
	public static int minimumParentheses(String pattern) {
	Stack<Character> st = new Stack<>();
	for(int i = 0; i<pattern.length() ; i++){
	if(pattern.charAt(i) == '('){
	st.push(pattern.charAt(i));
	}
	else if(pattern.charAt(i) == ')' && !st.empty() && st.peek() == '('){
	st.pop();
	} else {
	st.push(pattern.charAt(i));
	}
	}
	if(st.empty()){
		return 0 ;
	} else {
	return st.size();
	}
}
}
