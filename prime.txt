#include <stdio.h>
#include <math.h>
int main()
{
    int n;
    int temp;
    scanf("%d", &n);
    for (int i = 2; i < n; i++)
    {
        for (int j = 2; j <= sqrt(n); j++)
        {
            if (i % j == 0 && i != j)
            {
                temp = 0;
                break;
            }
            else
            {
                temp = i;
            }
        }
        if (temp == 0)
        {
        }
        else
        {
            printf("%d\n", temp);
        }
    }
}
